{# « Anterior #} {%- if page_atual > 1 -%} « {%- else -%} « {%- endif -%} {%- if total_paginas <= 7 -%} {# Mostra todas as páginas #} {%- for n in range(1, total_paginas + 1) -%} {%- if n == page_atual -%} {{ n }} {%- else -%} {{ n }} {%- endif -%} {%- endfor -%} {%- else -%} {# Truncamento: primeira + ... + janela em torno + ... + última #} {# Página 1 sempre #} {%- if page_atual == 1 -%} 1 {%- else -%} 1 {%- endif -%} {# Ellipsis antes #} {%- if _ini_meio > 2 -%} … {%- endif -%} {# Janela do meio #} {%- for n in range(_ini_meio, _fim_meio + 1) -%} {%- if n == page_atual -%} {{ n }} {%- else -%} {{ n }} {%- endif -%} {%- endfor -%} {# Ellipsis depois #} {%- if _fim_meio < total_paginas - 1 -%} … {%- endif -%} {# Última página sempre #} {%- if page_atual == total_paginas -%} {{ total_paginas }} {%- else -%} {{ total_paginas }} {%- endif -%} {%- endif -%} {# Próxima » #} {%- if page_atual < total_paginas -%} » {%- else -%} » {%- endif -%}
